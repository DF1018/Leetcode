```class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        max_count = 0
        seen = set(nums)
        for x in seen:
            if x-1 in seen:
                continue
            current = x
            while current in seen:
                current += 1
            max_count = max(max_count, current - x)
            if max_count == len(seen):  # 提前结束判断
                break
        return max_count
```
