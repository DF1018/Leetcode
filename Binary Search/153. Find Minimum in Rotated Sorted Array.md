class Solution:
    def findMin(self, nums: List[int]) -> int:
        if nums[0] <= nums[-1]:
            return nums[0]

        l, r = 0, len(nums) - 1
        res = float('inf')

        while l<=r:
            mid = l + ((r-l) // 2)
            if nums[mid] >= nums[0]: #代表在左半邊
                l = mid+1
            elif nums[mid] < nums[0]: #代表在右半邊
                res = min(res, nums[mid])
                r = mid-1
        return res
