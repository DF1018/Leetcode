### neetcode大神
```python
class Solution:
    def search(self, nums: list[int], target: int) -> int:
        l , r = 0 ,len(nums)-1
        while l<=r:
            mid =(l+r) // 2
            if target == nums[mid]:
                return mid

            #left sorted portion
            if nums[l] <= nums[mid]:
                if target > nums[mid] or target<nums[l]:
                    l = mid+1
                else:
                    r = mid-1
            #right sorted portion
            else:
                if target < nums[mid] or target > nums[r]:
                    r = mid -1
                else:
                    l = mid+1
                    # t > m and t < r
                
        return -1
```

```python
        while l<=r:
            m = (l+r)//2
            if nums[m]==target:
                return m
            elif nums[m]<nums[-1]:
                if nums[m]<target<=nums[-1]:
                    l=m+1
                else:
                    r=m-1
            else:
                if nums[0]<=target<nums[m]:
                    r=m-1
                else:
                    l=m+1
```

```python
class Solution:
    def search(self, nums: List[int], target: int) -> int:
        l , r = 0 , len(nums) - 1
        
        while l <= r:
            m = l + ((r-l) // 2)

            if nums[m] == target:
                return m

            if nums[l] <= nums[m]:
                if target >= nums[l] and target <= nums[m]:
                    r = m - 1
                else:
                    l = m + 1
            else:
                if target <= nums[r] and target >= nums[m]:
                    l = m + 1
                else:
                    r = m - 1
        return -1
```
