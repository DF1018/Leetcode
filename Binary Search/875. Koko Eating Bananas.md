```
class Solution:
    def minEatingSpeed(self, piles: list[int], h: int) -> int:
        k=0
        l , r = 1 , max(piles)
        res = r

        while l<=r:
            totel_sum = 0
            k = (l+r)//2
            
            for i in piles:
                totel_sum+=math.ceil(i/k)
            
            if totel_sum <= h:
                res=min(res,k)
                r=k-1
            else:
                l=k+1
            
        return res
```
