class Solution:
    def minEatingSpeed(self, piles: list[int], h: int) -> int:
        piles = sorted(piles)

        k=0
        l , r = 1 , piles[-1]
        _min = float("inf")

        while l<=r:
            totel_sum = 0
            k = (l+r)//2
            
            for i in piles:
                totel_sum+=math.ceil(i/k)
            
            if totel_sum < h:
                r=k-1
            elif totel_sum > h:
                l=k+1
            else:
                r=k-1
                _min=min(_min,k)
        return _min
