## neetcode
```
class Solution:
    def isSameTree(self, p: Optional[TreeNode], q: Optional[TreeNode]) -> bool:
        if not p and not q:
            return True
        if not p or not q or p.val != q.val:
            return False
        
        return (self.isSameTree(p.left, q.left) and
                self.isSameTree(p.right, q.right))
```

## DFS
```
class Solution:
    def isSameTree(self, p: Optional[TreeNode], q: Optional[TreeNode]) -> bool:
        # 如果兩個節點都是 None，樹相同
        if p is None and q is None:
            return True
        # 如果一個節點為 None 另一個不是，樹不相同
        elif p is None or q is None:
            return False

        is_left_same = self.isSameTree(p.left, q.left)
        is_right_same = self.isSameTree(p.right, q.right)


        return is_left_same and is_right_same and (p.val==q.val)
```

## BFS
```
class Solution:
    def isSameTree(self, p: Optional[TreeNode], q: Optional[TreeNode]) -> bool:
        queue = deque()
        queue.append((p,q))
        while queue:
            pNode, qNode = queue.popleft()
            if pNode and not qNode or not pNode and qNode:
                return False
            elif not pNode and not qNode:
                continue
            elif pNode.val != qNode.val:
                return False
            else:
                queue.append((pNode.left, qNode.left))
                queue.append((pNode.right, qNode.right))
        return True
```
