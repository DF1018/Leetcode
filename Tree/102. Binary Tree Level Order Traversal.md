```python
class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        q = collections.deque()
        q.append(root)
        
        res = []

        while q:
            qlen = len(q)
            temp = []
            for i in range(qlen):
                node = q.popleft()

                if node:
                    temp.append(node.val)
                    q.append(node.left)
                    q.append(node.right)

            if temp:
                res.append(temp)

          
        return res
```
