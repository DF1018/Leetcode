class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        q = deque()
        
        def inOrder(root):
            stack = []
            node = root

            while stack or node:
                # 遍歷所有左子節點
                while node:
                    stack.append(node)
                    node = node.left

                # 彈出 stack 頂的節點，將其值放入結果
                node = stack.pop()
                q.append(node.val)

                # 轉向右子樹
                node = node.right

        inOrder(root)
        for i in range(k):
            node_val = q.popleft()
        
        return node_val

class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        n = 0
        stack = []
        cur = root

        while cur or stack:
            while cur:
                stack.append(cur)
                cur = cur.left

            cur = stack.pop()
            n += 1
            if n == k:
                return cur.val
            cur = cur.right
