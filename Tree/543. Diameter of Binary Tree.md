```
class Solution:
    def diameterOfBinaryTree(self, root: Optional[TreeNode]) -> int:
        self.res = 0

        def dfs(curr):
            if not curr:
                return 0
            
            left = dfs(curr.left) #拿left該節點的高度
            right = dfs(curr.right) #拿right該節點的高度

            self.res = max(self.res,left+right)
            return 1 + max(left , right) #return出那個節點的最大高度 
            
        dfs(root)
        return self.res
```
