class Solution:
    def maxArea(self, height: List[int]) -> int:
        temp = 0
        res = 0
        l , r = 0 , len(height)-1

        while l < r:
            h=min(height[l],height[r])
            temp = h*(r-l)
            if height[l] < height[r]:
                l+=1
            elif height[l] > height[r]:
                r-=1
            else :
                l+=1

            res = max(res,temp)

        return res

class Solution:
    def maxArea(self, height: List[int]) -> int:
        l , r = 0 , len(height)-1
        res = 0
        while l < r:
            Length = min(height[l],height[r])
            Width = r-l
            res = max(res,Length*Width)
            
            if height[l] < height[r]:
                l+=1
            elif height[r] < height[l]:
                r-=1
            else:
                l+=1
                r-=1
        return res
