class Solution:
    def isPalindrome(self, s: str) -> bool:
        list_s = list(s)
        re_s = []


        for i in list_s:
            num = ord(i)
            if num >= 65 and num <= 90:
                num+=32
            if (num >=97 and num <=122) or (num>=48 and num<=57):   
                re_s.append(chr(num))

        for i in range(len(re_s)):
            left = i
            right = len(re_s)-i-1
            if not re_s[left] == re_s[right]:
                return False
        return True
```

class Solution:
    def isPalindrome(self, s: str) -> bool:
        newStr = ""

        for c in s:
            if c.isalnum():
                    newStr += c.lower()
        return newStr == newStr[::-1]
```

```
class Solution:
    def isPalindrome(self, s: str) -> bool:
        if len(s) == 1 or len(s) == 0:
            return True
        
        s = s.lower()
        new_s = ""
        
        for i in s:
            if ( ord(i) >= ord('a') and ord(i) <= ord('z') ) or ( ord(i) >= ord("0") and ord(i) <= ord("9") ):
                new_s+=i
        
        l , r = 0 , len(new_s)-1
        while l < r:
            l_point , r_point = new_s[l],new_s[r]
            if l_point == r_point:
                l+=1
                r-=1
            else:
                return False
        return True
```
