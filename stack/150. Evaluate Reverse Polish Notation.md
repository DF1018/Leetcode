class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        stack = []
        for c in tokens:
            if c == "+":
                stack.append(stack.pop()+stack.pop())
            elif c == "-":
                a,b = stack.pop() , stack.pop()
                stack.append(b - a)
            elif c == "*":
                stack.append(stack.pop()*stack.pop())
            elif c == "/":
                a,b = stack.pop() , stack.pop()
                stack.append(int(b / a))
            else:
                stack.append(int(c))
        return stack[0]

class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        stack = []
        hashmap = ["+","-","*","/"]
        
        for i in tokens:
            if i in hashmap:
                pop1=stack.pop()
                pop2=stack.pop()
                
                if i == "+":
                    stack.append(pop2 + pop1)
                elif i == "-":
                    stack.append(pop2 - pop1)
                elif i == "*":
                    stack.append(pop2 * pop1)
                elif i == "/":
                    stack.append(int(pop2 / pop1))
            else:
                stack.append(int(i))
                
        return stack[-1]
