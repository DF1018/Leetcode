class Solution:
    def isValid(self, s: str) -> bool:
        stack = []
        closeToOpen = { ")":"(" , "]":"[" , "}":"{" }

        for c in s:
            if c in closeToOpen:
                if stack and stack[-1] == closeToOpen[c]:
                    stack.pop()
                else:
                    return False
            else:
                stack.append(c)
        return True if not stack else False

class Solution:
    def isValid(self, s: str) -> bool:
        hashmap = {
            "{" : "}",
            "(" : ")",
            "[" : "]"
        }
        stack = []
        
        for cha in s:
            if cha in hashmap.keys():
                stack.append(cha)
            elif not len(stack) == 0:
                top = stack.pop()
                if hashmap[top] == cha:
                    continue
                else:
                    return False
            else:
                return False
                
                
            
        if len(stack) == 0:
            return True
        else:
            return False
