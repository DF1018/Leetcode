```
class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        stack = []
        res = []

        def backtrack(openN,closedN):
            if openN==closedN==n:
                res.append("".join(stack))
                return

            if openN<n:
                stack.append("(")
                backtrack(openN+1,closedN)
                stack.pop()
                                
            if closedN < openN:
                stack.append(")")
                backtrack(openN,closedN+1)
                stack.pop()

        backtrack(0,0)
        return res
```

```
start: open=0, close=0, stack=[]
 ├── "(" → open=1, close=0
 │    ├── "(" → open=2, close=0
 │    │    └── ")" → open=2, close=1
 │    │         └── ")" → open=2, close=2  ✅ "()()"
 │    └── ")" → open=1, close=1
 │         └── "(" → open=2, close=1
 │              └── ")" → open=2, close=2  ✅ "(())"
```
